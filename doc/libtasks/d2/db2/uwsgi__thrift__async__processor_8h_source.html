54  // Process message
55  worker* worker = worker::get();
56  m_handler.reset(new handler_type());
57  tdbg(get_string() << ": created handler " << m_handler.get() << std::endl);
58  m_handler->set_uwsgi_task(this);
59  m_handler->on_finish([this, worker, out_protocol] {

62  << "): " << m_handler->error_message() << std::endl);
63  write_thrift_error(std::string("Handler Error: ") + m_handler->error_message(),
64  m_handler->service_name(), out_protocol);
65  } else {
66  tdbg(get_string() << ": handler " << m_handler.get() << " finished with no error" << std::endl);
67  // Fill the response back in.
68  out_protocol->writeMessageBegin(m_handler->service_name(), T_REPLY, m_seqid);
69  m_handler->result_base().__isset.success = true;
70  m_handler->result_base().write(out_protocol.get());
71  out_protocol->writeMessageEnd();
72  out_protocol->getTransport()->writeEnd();
73  out_protocol->getTransport()->flush();
74  response().set_status("200 OK");
75  }

77  // Make sure we run in the context of a worker thread
78  if (error_code() == tasks_error::UNSET) {
79  worker->exec_in_worker_ctx([this](struct ev_loop*) { send_response(); });
80  }
81  // Allow cleanup now
82  enable_dispose();
83  });
84 
85  try {
86  std::string fname;
87  TMessageType mtype;
88  in_protocol->readMessageBegin(fname, mtype, m_seqid);
89  if (mtype != protocol::T_CALL && mtype != protocol::T_ONEWAY) {
90  write_thrift_error("invalid message type", m_handler->service_name(), out_protocol);
91  send_thrift_response();
92  } else if (fname != m_handler->service_name()) {
93  write_thrift_error("invalid method name", m_handler->service_name(), out_protocol);
94  send_thrift_response();
95  } else {

98  args->read(in_protocol.get());
99  in_protocol->readMessageEnd();
100  in_protocol->getTransport()->readEnd();
101  // Make sure the object is kept until the m_handler finishes
102  disable_dispose();

104  tdbg(get_string() << ": calling service on handler " << m_handler.get() << std::endl);
105  m_handler->service(args);
106  }
107  } catch (TException& e) {
108  write_thrift_error(std::string("TException: ") + e.what(), m_handler->service_name(), out_protocol);
109  send_thrift_response();
110  }
