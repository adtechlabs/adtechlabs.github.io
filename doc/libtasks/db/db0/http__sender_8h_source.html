40  bool handle_event(tasks::worker* worker, int events) {
41  bool success = true;
42  try {
43  if (EV_READ & events) {
44  m_response->read_data(socket());
45  if (m_response->done()) {
46  if (nullptr == m_handler) {
47  m_handler = std::make_shared<handler_type>();
48  }
49  success = m_handler->handle_response(m_response);
50  m_response->clear();
51  }
52  } else if (EV_WRITE & events) {
53  m_request->write_data(socket());
54  if (m_request->done()) {
55  // Reset the request buffer to be able to reuse the same object again
56  m_request->clear();
57  // Read the response
58  set_events(EV_READ);
59  update_watcher(worker);
60  }
61  }
62  } catch (tasks::tasks_exception& e) {
63  set_exception(e);
64  success = false;
65  }
66  return success;
67  }

75  inline void send(std::shared_ptr<http_request> request) {

77  const std::string& host = m_request->header("Host");
78  tdbg("http_sender: Sending request to " << host << std::endl);
79  // Find the worker if keepalive is used, and this is not the first request, or the object is reused to connect
80  // to a different host.
81  tasks::worker* worker = tasks::dispatcher::instance()->get_worker_by_task(this);
82  if (connected() && m_host != host) {
83  // Stop the watcher and close an existing connection.
84  tdbg("http_sender: Closing connection to " << m_host << ":" << m_port << std::endl);
85  stop_watcher(worker);
86  socket().close();
87  }
88  m_host = host;

91  set_events(EV_WRITE);

95  socket().connect(m_host, m_port);
